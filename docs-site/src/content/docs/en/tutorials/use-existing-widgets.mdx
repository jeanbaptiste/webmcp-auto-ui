This tutorial shows how to use the **built-in widgets** (stat, chart, timeline, etc.) provided by WebMCP Auto-UI in your Svelte application.

## Available widget types

WebMCP Auto-UI provides two categories of widgets:

### Simple widgets
- `stat` -- Key statistic (KPI, counter)
- `kv` -- Key-value pairs (properties, metadata)
- `list` -- Ordered list of items
- `chart` -- Bar chart
- `alert` -- Notification/alert
- `code` -- Code block with syntax highlighting
- `text` -- Text paragraph
- `actions` -- Action buttons
- `tags` -- Badges/tags

### Rich widgets
- `stat-card` -- Enhanced KPI with delta and colored variant
- `data-table` -- Sortable table with columns
- `timeline` -- Event chronology
- `profile` -- Profile card with avatar and stats
- `trombinoscope` -- Portrait grid
- `json-viewer` -- Interactive JSON tree
- `hemicycle` -- Parliamentary composition
- `chart-rich` -- Multi-series (bar, line, area, pie)
- `cards` -- Card grid
- `grid-data` -- Grid with cell highlights
- `sankey` -- Flow diagram
- `map` -- Interactive Leaflet map
- `log` -- Log stream
- `gallery` -- Image gallery with lightbox
- `carousel` -- Slide carousel
- `d3` -- D3.js visualizations
- `js-sandbox` -- Custom JavaScript sandbox

## Step 1: Import BlockRenderer

```svelte
<script lang="ts">
  import { BlockRenderer } from '@webmcp-auto-ui/ui';
  import { canvas } from '@webmcp-auto-ui/sdk/canvas';
</script>
```

## Step 2: Create widgets

Use the `canvas` store to add widgets:

```svelte
<script lang="ts">
  function addStatWidget() {
    canvas.addWidget('stat', {
      label: 'Total sales',
      value: '$12,450',
      trend: '+12%',
      trendDir: 'up',
    });
  }

  function addChartWidget() {
    canvas.addWidget('chart', {
      title: 'Monthly sales',
      bars: [
        ['January', 120],
        ['February', 190],
        ['March', 150],
      ],
    });
  }

  function addTableWidget() {
    canvas.addWidget('data-table', {
      title: 'Active customers',
      columns: [
        { key: 'name', label: 'Name' },
        { key: 'email', label: 'Email' },
        { key: 'status', label: 'Status' },
      ],
      rows: [
        { name: 'Alice Dupont', email: 'alice@example.com', status: 'Active' },
        { name: 'Bob Martin', email: 'bob@example.com', status: 'Suspended' },
        { name: 'Charlie Lenoir', email: 'charlie@example.com', status: 'Active' },
      ],
    });
  }
</script>

<button onclick={addStatWidget}>Add KPI</button>
<button onclick={addChartWidget}>Add chart</button>
<button onclick={addTableWidget}>Add table</button>
```

## Step 3: Display widgets

Use `BlockRenderer` to mount the canvas widgets:

```svelte
<div class="widgets-grid">
  {#each canvas.blocks as block (block.id)}
    <BlockRenderer
      id={block.id}
      type={block.type}
      data={block.data}
    />
  {/each}
</div>

<style>
  .widgets-grid {
    display: grid;
    grid-template-columns: repeat(auto-fit, minmax(300px, 1fr));
    gap: 1rem;
    padding: 1rem;
  }
</style>
```

### Interactive table

```svelte
<script lang="ts">
  function handleRowClick(row: Record<string, unknown>) {
    console.log('Row clicked:', row);
  }

  function addTable() {
    canvas.addWidget('data-table', {
      title: 'Article list',
      columns: [
        { key: 'title', label: 'Title', align: 'left' },
        { key: 'category', label: 'Category', align: 'center' },
        { key: 'views', label: 'Views', align: 'right' },
      ],
      rows: [
        { title: 'Introduction to Svelte', category: 'Tutorial', views: 1250 },
        { title: 'WebMCP Best Practices', category: 'Guide', views: 890 },
      ],
    });
  }
</script>

<div class="table-wrapper">
  {#each canvas.blocks as block (block.id)}
    {#if block.type === 'data-table'}
      <BlockRenderer
        id={block.id}
        type={block.type}
        data={block.data}
        oninteract={(type, action, payload) => {
          if (action === 'rowclick') handleRowClick(payload as Record<string, unknown>);
        }}
      />
    {:else}
      <BlockRenderer id={block.id} type={block.type} data={block.data} />
    {/if}
  {/each}
</div>
```

## Handling interactions

Some widgets emit events when the user interacts with them:

```svelte
<script lang="ts">
  function handleWidgetInteraction(
    type: string,
    action: string,
    payload: unknown
  ) {
    console.log(`Widget ${type} -> action: ${action}`, payload);

    // Examples:
    // type='data-table', action='rowclick', payload=row object
    // type='timeline', action='eventclick', payload=event object
    // type='cards', action='cardclick', payload=card object
    // type='gallery', action='imageclick', payload={image, index}
  }
</script>

<BlockRenderer
  id={block.id}
  type={block.type}
  data={block.data}
  oninteract={handleWidgetInteraction}
/>
```

## Dynamic updates

To modify a widget after creation:

```svelte
<script lang="ts">
  function updateStat() {
    canvas.updateBlock(blockId, {
      value: '$15,000',
      trend: '+20%',
      trendDir: 'up',
    });
  }

  function clearAllWidgets() {
    canvas.clearBlocks();
  }

  function removeOneWidget(blockId: string) {
    canvas.removeBlock(blockId);
  }
</script>

<button onclick={updateStat}>Update KPI</button>
<button onclick={clearAllWidgets}>Clear all</button>
```

## Theme and customization

Use `ThemeProvider` to customize appearance:

```svelte
<script lang="ts">
  import { ThemeProvider, LIGHT_TOKENS } from '@webmcp-auto-ui/ui';
</script>

<ThemeProvider theme="light" tokens={LIGHT_TOKENS}>
  <div class="widgets-grid">
    {#each canvas.blocks as block (block.id)}
      <BlockRenderer id={block.id} type={block.type} data={block.data} />
    {/each}
  </div>
</ThemeProvider>
```

## Use case: Dynamic dashboard

```svelte
<script lang="ts">
  import { BlockRenderer } from '@webmcp-auto-ui/ui';
  import { canvas } from '@webmcp-auto-ui/sdk/canvas';

  function buildDashboard(data: any) {
    canvas.clearBlocks();

    // Main KPI
    canvas.addWidget('stat-card', {
      label: 'Monthly revenue',
      value: data.revenue.toLocaleString('en-US'),
      unit: '$',
      trend: 'up',
      variant: 'success',
    });

    // Trend chart
    canvas.addWidget('chart-rich', {
      title: 'Trend',
      type: 'line',
      labels: data.months,
      data: [{ label: 'Revenue', values: data.values }],
    });

    // Transaction table
    canvas.addWidget('data-table', {
      title: 'Latest transactions',
      columns: [
        { key: 'date', label: 'Date' },
        { key: 'amount', label: 'Amount' },
        { key: 'status', label: 'Status' },
      ],
      rows: data.transactions,
    });
  }

  // Load on mount
  import { onMount } from 'svelte';
  onMount(async () => {
    const data = await fetch('/api/dashboard').then(r => r.json());
    buildDashboard(data);
  });
</script>

<div class="dashboard">
  {#each canvas.blocks as block (block.id)}
    <BlockRenderer id={block.id} type={block.type} data={block.data} />
  {/each}
</div>

<style>
  .dashboard {
    display: grid;
    grid-template-columns: repeat(auto-fit, minmax(350px, 1fr));
    gap: 1.5rem;
    padding: 2rem;
  }
</style>
```
